# map function

# filter function

items = [ ("Product 1", 100), ("Product 2", 90), ("Product 3", 120) ]

print(prices)

# list comprehensions

items = [ ("Product 1", 100), ("Product 2", 90), ("Product 3", 120) ]

prices = list(map(lambda item: item[1], items))
prices = [item[1] for item in items]

prices = list(filter(lambda item: item[1] >= 100 , items))
prices = [item for item in items if item[1] >= 100]

print(prices)

# Comprehensions

  1. we use comprehensions instead of map(), filter().
  2. for tuple comprehensions does not work, it creates a generator.
values = [x * 2 for x in range(5)] // list comprehensions
values = {x * 2 for x in range(5)} // set comprehensions
values = {x: x * 2 for x in range(5)} // set comprehensions
values = (x * 2 for x in range(5)) // tuple comprehensions, it creates a generator

# generator

  1. generator does not store value in memory.
  2. generator creates a new value in each iteration.
  3. we can not get the length of generator, because it generates value in each iteration.
  4. generator takes small amount of memory compared to other data structure.
use to get the size of list
from sys import getsizeof

values = (x * 2 for x in range(10000))
print(values) // output value : 120

values = [x * 2 for x in range(10000)]
print(values) // output value : 824464
